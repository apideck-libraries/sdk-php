use Apideck\Unify\Models\Components;
use Apideck\Unify\Models\Operations;

    ->setSecurity(
        '<YOUR_BEARER_TOKEN_HERE>'
    )
    ->setConsumerId('test-consumer')
    ->setAppId('dSBdXd2H6Mqwfg0atXHXYcysLJE9qyn1VwBtXHX')
    ->build();

$request = new Operations\CrmCustomObjectsAddRequest(
    serviceId: 'salesforce',
    objectId: '<id>',
    customObject: new Components\CustomObjectInput(
        name: 'project',
        fields: [
            new Components\CustomObjectFields(
                name: 'name',
                value: 'string',
            ),
        ],
        passThrough: [
            new Components\PassThroughBody(
                serviceId: '<id>',
                extendPaths: [
                    new Components\ExtendPaths(
                        path: '$.nested.property',
                        value: [
                            'TaxClassificationRef' => [
                                'value' => 'EUC-99990201-V1-00020000',
                            ],
                        ],
                    ),
                ],
            ),
        ],
    ),
);

$response = $sdk->crm->customObjects->create(
    request: $request
);

if ($response->createCustomObjectResponse !== null) {
    // handle response
}
